# 条件查询

```sql
# 条件查询
/*
语法：
	select 
		查询列表
	from
		表明
	where
		筛选条件
分类：1.条件表达式筛选（条件运算符：> < = != <> <= >=）
2.逻辑表达式筛选 && || |  and or not
3.模糊查询 like 、between and 、in 、 is null
*/
SELECT 
  * 
FROM
  employees 
WHERE salary > 12000

SELECT 
    last_name,
    department_id 
  FROM
    employees 
  WHERE `department_id` <> 90 
  
    SELECT 
      last_name,
      salary,
      commission_pct 
    FROM
      employees 
    WHERE salary >= 10000 
      AND salary <= 20000 
      
 #查询员工名中包含a的员工信息
      SELECT 
        * 
      FROM
        employees 
      WHERE last_name LIKE '%a%' 
      #   %号表示任意字符

```
